# Connection Cubes

Connection cubes are currently exposed only as [python custom members](./pythonmembers.html). You will need to utilize python custom members in order to manipulate connection cubes.

Some python modules will be needed for creating and using connection cubes

```
static void LoadPythonModules()
{
    connectionCubeModule = Python.Module.Import("ConnectionCube");
    builtinsModule = Python.Module.Import("builtins");
    outputModule = Python.Module.Import("Output");
}

```

Connection cubes are created by providing a list of tuples containing member numbers and the member end of interest:

```
static bool CreateConnectionCube(/* 1st element is a member number
                                    2nd element is the member end*/
                                List<Tuple<int, int>> membersAndEnds,
                                string cubeName = "")
{
    dynamic ends = builtinsModule.list();
    foreach (Tuple<int, int> memEnd in membersAndEnds)
    {
        dynamic pyMemEnd = builtinsModule.list();
        pyMemEnd.append(memEnd.Item1);
        pyMemEnd.append(memEnd.Item2);

        ends.append(builtinsModule.tuple(pyMemEnd));
    }

    if(cubeName.Length > 0)
        return connectionCubeModule.AddNamed(ends, cubeName);
    else
        return connectionCubeModule.Add(ends);
}

```

Retrieving all connection cubes:

```
static object GetAllConnectionCubes()
{
    dynamic allCubes = connectionCubeModule.GetAllConnectionCubes();

    return allCubes;
}

```

Retrieving a specific connection cube:

```
static object GetCubeByIndex(int index)
{
    dynamic cubes = GetAllConnectionCubes();
    int cubeLen = cubes.__len__();

    return cubes.__getitem__(index);
}

```

Deleting a connection cube:

```
static void DeleteCube(int cubeIndex)
{
    dynamic allCubes = connectionCubeModule.GetAllConnectionCubes();
    if (cubeIndex >= allCubes.__len__())
    {
        return;
    }

    connectionCubeModule.Delete(allCubes.__getitem__(cubeIndex).cube_number);
}

```

```
static void DeleteAllCubes()
{
    dynamic allCubes = connectionCubeModule.GetAllConnectionCubes();
    for(int i = 0; i < allCubes.__len__(); i++)
    {
        connectionCubeModule.Delete(allCubes.__getitem__(i).cube_number);
    }
}

```

Cube reports can be generated by supplying a list of connection cube numbers. Each cube will have it's own report. If an output path is not supplied, reports will be saved in the output directory of the data direcory:

```
static void GenerateCubeReport(List<dynamic> cubes)
{
    dynamic index_list = builtinsModule.list();
    foreach(dynamic cube in cubes)
    {
        index_list.append(cube.cube_number);
    }

    try
    {
        // NOTE: If providing a path for the generated report, that path
        //       must already exist.
        string report_path = Path.Join(Path.GetTempPath(), "cube_report");
        Directory.CreateDirectory(report_path);

        dynamic dest = outputModule.Destination(report_path);

        connectionCubeModule.GenerateReport(index_list, dest);
    }
    catch (Exception e)
    {
        Console.WriteLine(e.ToString());
    }
}

```

Putting it all together. Note that this example makes use of the SampleJob that comes with an SDS2 installation.

```
static void Run()
{
    DataDirectory.Open(DataDirectory.Default);
    Job job = OpenNamedJob("Default Repository", "Building_101j");

    using (var reader = new ReadOnlyTransaction(job))
    {
        reader.RefreshTable(Table.Member);
    }

    LoadPythonModules();

    dynamic ends = builtinsModule.list();

    var membersAndEnds = new List<Tuple<int, int>>();
    membersAndEnds.Add(new Tuple<int, int>(45, 1));
    membersAndEnds.Add(new Tuple<int, int>(46, 0));
    membersAndEnds.Add(new Tuple<int, int>(49, 0));
    membersAndEnds.Add(new Tuple<int, int>(50, 1));
    membersAndEnds.Add(new Tuple<int, int>(30, 1));

    if (CreateConnectionCube(membersAndEnds, "MyCube"))
    {
        Console.WriteLine("Cube created successfully.");
    }

    // Create connection cube containing the group member
    membersAndEnds.Clear();
    membersAndEnds.Add(new Tuple<int, int>(9, 1));
    membersAndEnds.Add(new Tuple<int, int>(42, 0));
    membersAndEnds.Add(new Tuple<int, int>(49, 1));
    membersAndEnds.Add(new Tuple<int, int>(79, 1));

    CreateConnectionCube(membersAndEnds, "GM_Cube");

    // Create connection cube involving a horizontal brace
    membersAndEnds.Clear();
    membersAndEnds.Add(new Tuple<int, int>(55, 0));
    membersAndEnds.Add(new Tuple<int, int>(25, 0));
    CreateConnectionCube(membersAndEnds, "Brace_Cube");

    dynamic allCubes = GetAllConnectionCubes();
    Console.WriteLine("Found {0} connection cubes in the job", allCubes.__len__());

    dynamic cube = GetCubeByIndex(allCubes.__len__()-1);

    List<dynamic> reportCubes = new List<dynamic>();
    for (int index = 0; index < allCubes.__len__(); index++)
    {
        reportCubes.Add(GetCubeByIndex(index));
    }

    GenerateCubeReport(reportCubes);

    DeleteCube(2);
    DeleteAllCubes();
}
```
